# Operators are of 4 types
# Arithmetic : + , - , *, / , ** , // , % (add,sub,mul,div,exponential,floordivision,modulus)
# Assignment : =, +=, -=, *=, /= (Mostly used)
# Comparision : ==, !=, >, <, >=, <=
# Logical : and , or , not
# Bitwise : &, |, ^, ~, <<, >>
# Identity : is 
# Membership : in, not in
